# 189. 轮转数组（中等）
## 题目：
给定一个整数数组 `nums`，将数组中的元素向右轮转 `k` 个位置，其中 `k` 是非负数。
## 题解：
### 方法一：额外数组
将要移动的部分另存，然后移动数组。
```c++
class Solution {
public:
    void rotate(vector<int>& nums, int k) {
        int n=nums.size();
        k=k%n;
        int cnt=0;
        vector<int> temp(nums.begin()+n-k,nums.end());
        int j=n-k-1;
        for(int i=n-1;i>k-1;i--){
            nums[i]=nums[j];
            j--;
        }
        for(int i=0;i<k;i++){
            nums[i]=temp[i];
        }
    }
};
```

```c++
class Solution {
public:
    void reverse(vector<int>& nums, int start, int end){
        while(start<end){
            swap(nums[start], nums[end]);
            start++;
            end--;
        }
    }
    void rotate(vector<int>& nums, int k) {
        int n=nums.size();
        k=k%n;
        reverse(nums, 0, n-1);
        reverse(nums, 0, k-1);
        reverse(nums, k, n-1);
    }
};
```
